# GIAI ĐOẠN 1: BUILD (Sử dụng SDK để biên dịch code)
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copy file sln và csproj vào trước để restore (tối ưu cache)
COPY ["APT.sln", "./"]
COPY ["APT/APT.csproj", "APT/"]
RUN dotnet restore

# Copy toàn bộ code còn lại và build
COPY . .
WORKDIR "/src/APT"
RUN dotnet publish "APT.csproj" -c Release -o /app/publish /p:UseAppHost=false

# GIAI ĐOẠN 2: RUN (Sử dụng Runtime nhẹ để chạy app)
FROM mcr.microsoft.com/dotnet/aspnet:8.0
WORKDIR /app
COPY --from=build /app/publish .

# Cấu hình biến môi trường để Railway nhận diện cổng
ENV ASPNETCORE_URLS=http://+:8080
EXPOSE 8080

# Lệnh khởi chạy file dll (Đảm bảo tên file khớp với tên project của anh)
ENTRYPOINT ["dotnet", "APT.dll"]
